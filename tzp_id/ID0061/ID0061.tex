\tzpsection{Derivation Layer}
\paragraph{Primary Equation.}
\begin{equation}
S_A = -\mathrm{Tr}(\rho_A \log \rho_A).
\end{equation}

\paragraph{Primary Derivation.}
Let $\rho$ be the density matrix of a composite system partitioned into subsystems $A$ and $B$.

The reduced density matrix is:
\begin{equation}
\rho_A = \mathrm{Tr}_B(\rho).
\end{equation}

The entanglement entropy is defined as:
\begin{equation}
S_A = -\mathrm{Tr}(\rho_A \log \rho_A).
\end{equation}

\paragraph{Interpretation.}
Entropy measures quantum correlation between subsystems.
