\begin{equation}

\begin{equation}
\dot{L}

I_T\dot{\omega},
\end{equation}
assuming $I_T$ is constant over the interval.

Since $\dot{L}=\tau_{\mathrm{em}}$, we obtain:
\begin{equation}
\dot{\omega}(t)
=
\frac{\tau_{\mathrm{em}}(t)}{I_T}.

\begin{equation}

\omega(t)
=
\omega(0)
+
\frac{1}{I_T}
\int_0^t
\tau_{\mathrm{em}}(t')
\,\mathrm{d}t' .

\end{equation}
