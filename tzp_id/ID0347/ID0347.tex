\begin{equation}
P(H_i\mid D)

Likewise,

\begin{equation}
P(H_j\mid D)
=
\frac{P(D\mid H_j)P(H_j)}
{P(D)} .
\end{equation}

Taking the ratio cancels the evidence denominator $P(D)$:

\begin{equation}
\frac{P(H_i\mid D)}
{P(H_j\mid D)}

\frac{P(D\mid H_i)P(H_i)}
{P(D\mid H_j)P(H_j)} .
\end{equation}

Rearranging gives:

\begin{equation}
\boxed{
\frac{P(H_i\mid D)}
{P(H_j\mid D)}
=
\frac{P(D\mid H_i)}
{P(D\mid H_j)}
