\paragraph{Primary Equation.}
\begin{equation}

\paragraph{Trace-Preserving Constraint.}
\begin{equation}
\sum_k E_k^\dagger E_k = I .
\end{equation}

\paragraph{Primary Derivation.}
A completely positive map admits a Kraus representation:
\begin{equation}
\mathcal{E}(\rho)=\sum_k E_k\rho E_k^\dagger .
\end{equation}

Trace preservation requires
\begin{equation}
\operatorname{Tr}(\mathcal{E}(\rho))=\operatorname{Tr}(\rho).
\end{equation}

Using cyclicity of trace,
\begin{equation}
\operatorname{Tr}(\mathcal{E}(\rho))
=
\operatorname{Tr}\left(
\rho\sum_k E_k^\dagger E_k
\right).
\end{equation}

This equals $\operatorname{Tr}(\rho)$ for all $\rho$ precisely when
\begin{equation}
\boxed{
\sum_k E_k^\dagger E_k = I .
}
\end{equation}

\paragraph{Interpretation.}
CPTP maps preserve both positivity and total probability, making them valid quantum-state transformations.
