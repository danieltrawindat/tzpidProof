\tzpsection{Derivation Layer}
\paragraph{Primary Equation.}
\begin{equation}
C(\phi_1,\phi_2)
=
\frac{
\left|\langle \phi_1,\phi_2\rangle\right|
}{
\lVert \phi_1\rVert\,\lVert \phi_2\rVert
}.
\end{equation}

\paragraph{Primary Derivation.}
Let $\phi_1$ and $\phi_2$ be two field configurations on $M$.  
Their raw overlap is measured by the inner product:

\begin{equation}
\langle \phi_1,\phi_2\rangle .
\end{equation}

To remove dependence on field magnitude, normalize by the product of the two norms:

\begin{equation}
\lVert \phi_1\rVert\,\lVert \phi_2\rVert .
\end{equation}

Taking the absolute value gives a nonnegative coherence score:

\begin{equation}
\boxed{
C(\phi_1,\phi_2)
=
\frac{
\left|\langle \phi_1,\phi_2\rangle\right|
}{
\lVert \phi_1\rVert\,\lVert \phi_2\rVert
}.
}
\end{equation}

\paragraph{Interpretation.}
The field coherence metric measures normalized alignment between two field states. Values near $1$ indicate strong coherence; values near $0$ indicate decoherence or orthogonality.
